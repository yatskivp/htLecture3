   { scores: { $elemMatch: { $and : [
			{score : {$gt : 87}},

	] } } }

	{$unwind: "$scores"}, 
	{$match: {"scores.type": "exam"}}, 
	{$match: {"scores.score" : {$gt : 90}}},
	{$project: {_id : 0, name : 1, scores : 1}}
);
